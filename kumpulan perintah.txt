// eksport semua kode
Get-ChildItem -Recurse -File | Where-Object { $_.FullName -notmatch 'node_modules|\.git|dist|build|venv' -and $_.Extension -match '\.(js|ts|py|c|cpp|cs|html|css|php|java)$' } | ForEach-Object { "--- START FILE: $($_.FullName) ---`n"; Get-Content $_.FullName; "`n--- END FILE ---`n`n" } | Out-File -FilePath "Project_Semua_Kode.txt" -Encoding utf8

# 1. Tambahkan semua perubahan file
git add .

# 2. Buat catatan perubahan (Commit)
git commit -m "ganti baru"

# 3. Kirim ke GitHub
git push origin main
